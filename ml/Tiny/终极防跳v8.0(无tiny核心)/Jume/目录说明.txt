+ZJstart.sh  -> 启动终极防跳
+ZJcheck.sh  -> 检测终极防跳

同时再ZJstart.sh中添加如下代码
#################################
echo ""
echo "---------------------"
killall -9 tiny >/dev/null 2>&1
/system/xbin/Jume/tiny -c /system/xbin/Jume/tiny.conf
if [[ `ps|grep tiny|grep -v grep` != "" ]]
then  echo "o tiny 已运行"
else echo "x tiny未运行";fi
